200  def determine_amount(self):
201  """Computes the correct value of the deformation amount.
202 
203  The deformation amount of the structure is first determined as the
204  minimum deformable length of the deformingComps.
205  Then connections are taken into account.
206  Finally, the minimum amount is saved in self.amount.
207 
208  Args:
209  nothing is taken
210  Returns:
211  nothing is returned
212  Raises:
213  nothing is raised
214  """
215  # loop over the components and get the minimum deformable_length
216  amount = min(component.deformable_length()
217  for component in self.deformingComps)
218 
219  # Get the nodes that are leading the deformation
220  # i.e. the right node of each deforming component
221  deformationLeadingNodes = [component.rightNode
222  for component in self.deformingComps]
223 
224  # Calculate the deformation that is allowed by the deforming
225  # cross-components
226  amountCrossComponents = self.cross_components_amount(
227  deformationLeadingNodes)
228 
229  # Get the minimum amount
230  if amountCrossComponents is not None:
231  self.amount = min(amount, amountCrossComponents)
232  # if a connection is limiting the amount of deformation
233  if amountCrossComponents < amount:
234  self.amount = amountCrossComponents
235  # in the next defomation step, every component might deform
236  self.keep = False
237  # otherwise
238  else:
239  self.amount = amount
240  # the components that are still deformable should keep on
241  # deforming
242  self.keep = True
243  else:
244  self.amount = amount
245  self.keep = True
